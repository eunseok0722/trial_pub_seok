레이어 팝업 (Basic)

- pop01 : 팝업 ID Name
- 팝업오픈시 class="btn_pop ID Name_open
- 팝업닫기시 class="btn_pop_close ID Name_close

<a href="#pop01" class="btn_ty btn_m btn_pop pop01_open">팝업 버튼</a>
<!-- popup -->
<section class="modal" id="pop01">
    <div class="modal_wrap">
        <!-- modal_header -->
        <div class="modal_header">
            <h2>팝업타이틀</h2>
        </div>
        <!-- //modal_header -->
        <!-- modal_body -->
        <div class="modal_body pop_ty">
            <article class="cont">
                팝업내용
            </article>
        </div>
        <!-- //modal_body -->
        <button class="pop01_close btn_pop_close">팝업 닫기</button>
    </div>
</section>
<!-- //popup -->

- 팝업 자동 오픈
<script type="text/javascript">
jQuery(function($){
    $('#pop01').popup({
        autoopen: true,
    });
});

- 팝업 함수
<script type="text/javascript">
jQuery(function($){
    $('#pop01').popup('show');   // 열림
    $('#pop01').popup('hide');   // 닫음
	$('#pop01').popup({
        onopen:function(){
            alert('open 함수');
        }
        onclose:function(){
            alert('close 함수');
        }
    });
});
</script>

<script type="text/javascript">
jQuery(function($){
	/* 다른 페이지 탭 열림 : 버튼 class / 탭 nav id 확인 */
	if(location.href.indexOf("?ns1") > -1){
		setTimeout(function() {
			$('.btn_ns1').trigger('click');
		}, 10);	
	}	
	/* 같은 페이지 탭 열림 : 버튼 class / 탭 nav id 확인 */
	$('.btn_ns1').click(function(){
		$('#ns1 > a').trigger('click');
		return false;
	});

	/* 다른 페이지 아코디언 열림 : 버튼 class / 아코디언 타겟 id 확인 */
	if(location.href.indexOf("?ts1") > -1){
		setTimeout(function() {
			$('.btn_ts1').trigger('click');
		}, 10);	
	}
	/* 같은 페이지 아코디언 열림 : 버튼 class / 아코디언 타겟 id 확인 */
	$('.btn_ts1').click(function(){
		var tgt = $('#ts1');
		var tgtFocus = $('#ts1 > a')
		tgt.addClass('on');
		tgt.css('height', tgt.find('.view').innerHeight() + 68);
		$('html, body').stop().animate({scrollTop:tgt.offset().top - 190}, 600);
		tgtFocus.focus();
		return false;
	});
});
</script>

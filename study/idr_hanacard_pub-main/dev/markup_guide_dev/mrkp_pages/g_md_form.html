$('#test01').selectBox();                   // Show
$('#test01').selectBox('destroy');          // Hide

$('#test01').selectBox('enable');           // Enable
$('#test01').selectBox('disable');          // Disable

$('#test01').selectBox('refresh');          // 새로고침

$('#test01').selectBox('value',1);          // value 체크
$('#test01').selectBox('value',2);

$('#test01').selectBox('options',{          // options 변경
    '1' : 'Value 1',
    '2' : 'Value 2',
    '3' : 'Value 3',
    '4' : 'Value 4',
    '5' : 'Value 5'
});

$('#append').click(function(){
    $('#test01').append("<option value='10'>추가</option>").selectBox('refresh');   // option 추가 시
});

$('#remove').click(function(){
    $("#test01 option[value='10']").remove();
    $('#test01').selectBox('refresh');   // option 제거 시
});
